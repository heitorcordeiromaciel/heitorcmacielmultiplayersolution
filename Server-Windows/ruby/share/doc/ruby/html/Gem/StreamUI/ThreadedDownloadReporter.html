def fetch(file_name, *args)
  if @file_name.nil?
    @file_name = file_name
    locked_puts "Fetching #{@file_name}"
  end
end

    Tells the download reporter that the file_name is being fetched. The other arguments are ignored.

def locked_puts(message)
  MUTEX.synchronize do
    @out.puts message
  end
end
